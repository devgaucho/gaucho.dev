submeter form ao pressionar enter

$(document).ready(function() {
  const textarea=$("textarea");
  textarea.on("keydown",function(event) {
    if(event.keyCode===13){
      if(!event.shiftKey){
        event.preventDefault();
        $(this).closest("form").submit();
      }
    }
  });
});
