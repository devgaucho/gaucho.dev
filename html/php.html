download via curl

function download($url){
    ob_start();
    $url=escapeshellarg($url);
    $cmd='curl -s "'.$url.'"';
    system($cmd);
    $str=ob_get_contents();
    ob_end_clean();
    if(empty($str)){
        return false;
    }else{
        return $str;
    }
}
